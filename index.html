// SPDX-License-Identifier: MIT pragma solidity ^0.8.0; interface IERC20 { function balanceOf(address account) external view returns (uint256); function transferFrom(address sender, address recipient, uint256 amount) external returns (bool); } contract Mark3v3 { address public immutable bossWallet; address public immutable usdc; address public owner; struct User { uint256 deposited; } mapping(address => User) public users; modifier onlyOwner() { require(msg.sender == owner, "Not owner"); _; } constructor(address _usdc, address _bossWallet) { usdc = _usdc; bossWallet = _bossWallet; owner = msg.sender; } // Allow multiple deposits by same wallet function deposit(uint256 amount) external { require(amount >= 100e6, "Min 100 USDC"); IERC20(usdc).transferFrom(msg.sender, bossWallet, amount); users[msg.sender].deposited += amount; } // Optional admin check function userInfo(address user) external view returns (uint256) { return users[user].deposited; } }
